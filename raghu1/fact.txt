#include <stdio.h>
double factorial(int num) {
    double fact = 1;
    for (int i = 1; i <= num; i++) {
        fact *= i;
    }
    return fact;
}

int main() {
    int n;
    double sum = 0.0;

    printf("Enter the value of n: ");
    scanf("%d", &n);

    for (int i = 1; i <= n; i++) {
        sum += (double)i / factorial(i);  
    }

    printf("Sum of the series 1/1! + 2/2! + ... + %d/%d! = %.6lf\n", n, n, sum);

    return 0;
}

OUTPUT:

Enter the value of n: 8
Sum of the series 1/1! + 2/2! + ... + 8/8! = 2.718254
