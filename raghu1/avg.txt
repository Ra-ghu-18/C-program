#include <stdio.h>
int main() {
    int num1, num2, num3;  
    int sum;               
    float average;         
    printf("Enter three integers: ");
    scanf("%d %d %d", &num1, &num2, &num3);
    sum = num1 + num2 + num3;
    average = (float)sum / 3;
    printf("The average of the three integers is: %.2f\n", average);
    return 0;
}

OUTPUT:

Enter three integers: 46
35
78
The average of the three integers is: 53.00
