#include <stdio.h>

void convertToOctal(int num) {
    int octal[50], i = 0;

    while (num != 0) {
        octal[i++] = num % 8;
        num /= 8;
    }

    printf("Octal equivalent: ");
    for (int j = i - 1; j >= 0; j--)
        printf("%d", octal[j]);
    printf("\n");
}

int main() {
    int num;
    printf("Enter a decimal number: ");
    scanf("%d", &num);

    convertToOctal(num);
    return 0;
}

OUTPUT:

Enter a decimal number: 12.8
Octal equivalent: 14
