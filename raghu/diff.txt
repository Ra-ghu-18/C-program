#include <stdio.h>

int main() {
    int n, i, num;
    int smallest, largest;

    printf("Enter the number of elements: ");
    scanf("%d", &n);

    printf("Enter the numbers:\n");
    for (i = 0; i < n; i++) {
        scanf("%d", &num);
        if (i == 0) {
            smallest = largest = num;
        } else {
            if (num < smallest) {
                smallest = num;
            }
            if (num > largest) {
                largest = num;
            }
        }
    }

    int difference = largest - smallest;
    printf("The difference between the largest and smallest number is: %d\n", difference);

    return 0;
}
