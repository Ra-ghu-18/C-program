#include <stdio.h>

void fibonacci(int n) {
    int a = 0, b = 1, next, i;

    for (i = 1; i <= n; i++) {
        printf("%d ", a);
        next = a + b;
        a = b;
        b = next;
    }
    printf("\n");
}

int main() {
    int n;

    printf("Enter the number of terms for Fibonacci sequence: ");
    scanf("%d", &n);

    fibonacci(n);
    return 0;
}

OUTPUT:

Enter the number of terms for Fibonacci sequence: 5 
0 1 1 2 3
